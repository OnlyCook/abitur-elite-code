 // Operation der zweithöchsten Priorität
 if (zweithoechste > -1) {

 if (op.getIndikation().getPriorität() ==

 while(positionLinks < positionRechts) {

 positionLinks++;

0

 positionRechts--;
 if(positionLinks < positionRechts) {
 karten = tauscheKarte(karten,positionLinks, 
 positionRechts);
 }
 }

 karten = tauscheKarte(karten, positionLinks, ende);
 }
 return positionLinks;
 }

 checkValue = readChar();

 Serial com = new Serial("COM1", 9600, 8, 1, 0);

public List<Lehrkraft> sucheVertretungslehrer(Date d,
 int stunde, Lehrkraft l) {
 List<Lehrkraft> vertretungsLehrer = new List<Lehrkraft>();
 // betroffene Unterrichtsstunde ermitteln
 Unterrichtsstunde uStunde = 
 sucheUnterrichtsstunde(d, stunde, l);
 // alle Lehrkräfte ermitteln, die in an diesem Tag 
 // und in dieser Stunde keinen Unterricht und keine
 // Vertretung haben
 for (Lehrkraft lehrkraft : lehrkraefte) {
 if (! lehrkraft.hatUnterricht(d, stunde) && 
 ! lehrkraft.hatVertretung(d, stunde)) {
 vertretungsLehrer.add(lehrkraft);
 }
 }
 // alle Lehrkräfte ermitteln, die an diesem Tag und in 
 // dieser Stunde keinen Unterricht haben und in der Klasse 
 // unterrichten
 if (vertretungsLehrer.size() > 1) {
 Klasse klasse = uStunde.getKlasse();
 List<Lehrkraft> ersteWahl = 
 sucheVertretungslehrer(vertretungsLehrer, klasse);
 if (ersteWahl.size() > 0) {
 vertretungsLehrer = ersteWahl;
 } else {
 // alle Lehrkräfte ermitteln, die in an diesem Tag und 
 // in dieser Stunde keinen Unterricht haben und das
 // Fach unterrichten
 Fach fach = uStunde.getFach();
 List<Lehrkraft> zweiteWahl = 
 sucheVertretungslehrer(vertretungsLehrer, fach);
 if (zweiteWahl.size() > 0 )
 vertretungsLehrer = zweiteWahl;
 }
 }
 return vertretungsLehrer;
} 

public void hinzufuegenVertretung(Date d,
 Unterrichtsstunde s) {
 boolean positionGefunden = false;
 int index = 0;
 for (; index < vertretungen.size() && !positionGefunden; 
 index++) {
 Vertretung v = vertretungen.get(index);
 if (v.getDatum().compareTo(d) > 0
 || ((v.getDatum().compareTo(d) == 0)
 && (v.getVertretungsstunde().getStunde() > 
 s.getStunde()))) {
 positionGefunden = true;
 }
 }
 if (positionGefunden)
 index--;
 vertretungen.add(index, new Vertretung(d, this, s));
}

public String readCard() {
String response = "";
char checkValue;
char[] data = new char[6];
while(readChar() != 0x02); // warte auf STX
// lies Zeichen von der Schnittstelle
for(int i = 0; i < data.length; i++) {
data[i] = readChar();
}
// lies Pruefzeichen

private Film film;
public Vorstellung(Kinosaal kinosaal, Film film,

this.kinosaal = kinosaal;
this.film = film;
sitzbelegung = new boolean [this.kinosaal.getReihen()]

public int sucheBesteFreieSitze(int anzahl) {
int ersterFreierSitz = 0;
int anzahlFreieSitze = 0;
for (int reihe=sitzbelegung.length-1;reihe>=0
&& anzahlFreieSitze < anzahl;
reihe--) {
anzahlFreieSitze = 0;
for (int sitz = 0; sitz < sitzbelegung[reihe].length
&& anzahlFreieSitze < anzahl;
sitz++) {
if (sitzbelegung[reihe][sitz]) {
anzahlFreieSitze++;
if (anzahlFreieSitze == anzahl) {
ersterFreierSitz = (reihe+1) * 100 + (sitz+1)

anzahlFreieSitze = 0;
}
}
